## Overview

Blumbaben provides flexible toolbar positioning through the `getPosition` configuration option. You can use the default positioning or create custom positioning logic to suit your UI needs.

## The ToolbarPosition Type

Position coordinates are represented as:

```typescript
type ToolbarPosition = {
    // Horizontal position in pixels from the left edge of the viewport
    x: number;
    
    // Vertical position in pixels from the top edge of the viewport
    y: number;
};
```

<Info>
Positions are **viewport-relative** using `position: fixed` in CSS, not relative to the input element.
</Info>

## Default Positioning

By default, the toolbar appears **5 pixels below** the focused input element, aligned to its left edge:

```typescript
const defaultGetPosition = (element: TextInputElement): ToolbarPosition => {
    const rect = element.getBoundingClientRect();
    return {
        x: rect.left,
        y: rect.bottom + 5,
    };
};
```

## Custom Positioning

Provide a custom `getPosition` function in the toolbar configuration:

```typescript
type ToolbarConfig = {
    getPosition?: (element: TextInputElement) => ToolbarPosition;
    // ... other config options
};
```

### Position Above Element

```typescript
import { useFormattingToolbar } from 'blumbaben';

function MyComponent() {
    const { getInputProps, getToolbarProps, isVisible } = useFormattingToolbar({
        getPosition: (element) => {
            const rect = element.getBoundingClientRect();
            return {
                x: rect.left,
                y: rect.top - 50, // 50px above the element
            };
        },
    });

    return (
        <>
            <textarea {...getInputProps()} />
            {isVisible && <div {...getToolbarProps()}>Toolbar</div>}
        </>
    );
}
```

### Center Align Toolbar

```typescript
const { getInputProps } = useFormattingToolbar({
    getPosition: (element) => {
        const rect = element.getBoundingClientRect();
        const toolbarWidth = 200; // Your toolbar's width
        
        return {
            x: rect.left + (rect.width - toolbarWidth) / 2,
            y: rect.bottom + 5,
        };
    },
});
```

### Right Align Toolbar

```typescript
const { getInputProps } = useFormattingToolbar({
    getPosition: (element) => {
        const rect = element.getBoundingClientRect();
        const toolbarWidth = 200;
        
        return {
            x: rect.right - toolbarWidth,
            y: rect.bottom + 5,
        };
    },
});
```

## Common Positioning Patterns

### Follow Cursor

Position the toolbar near the text cursor instead of the element:

```typescript
import { useFormattingToolbar } from 'blumbaben';

function CursorFollowingToolbar() {
    const { getInputProps, getToolbarProps, isVisible } = useFormattingToolbar({
        getPosition: (element) => {
            const rect = element.getBoundingClientRect();
            const selection = window.getSelection();
            
            if (selection && selection.rangeCount > 0) {
                const range = selection.getRangeAt(0);
                const rangeRect = range.getBoundingClientRect();
                
                return {
                    x: rangeRect.left,
                    y: rangeRect.bottom + 5,
                };
            }
            
            // Fallback to element position
            return {
                x: rect.left,
                y: rect.bottom + 5,
            };
        },
    });

    return (
        <>
            <textarea {...getInputProps()} />
            {isVisible && <div {...getToolbarProps()}>Toolbar</div>}
        </>
    );
}
```

### Constrain to Viewport

Ensure the toolbar stays within the visible viewport:

```typescript
const { getInputProps } = useFormattingToolbar({
    getPosition: (element) => {
        const rect = element.getBoundingClientRect();
        const toolbarWidth = 300;
        const toolbarHeight = 50;
        
        let x = rect.left;
        let y = rect.bottom + 5;
        
        // Constrain to viewport width
        if (x + toolbarWidth > window.innerWidth) {
            x = window.innerWidth - toolbarWidth - 10;
        }
        if (x < 10) {
            x = 10;
        }
        
        // Constrain to viewport height
        if (y + toolbarHeight > window.innerHeight) {
            // Position above element instead
            y = rect.top - toolbarHeight - 5;
        }
        if (y < 10) {
            y = 10;
        }
        
        return { x, y };
    },
});
```

### Fixed Position

Keep the toolbar at a fixed location regardless of element position:

```typescript
const { getInputProps } = useFormattingToolbar({
    getPosition: () => {
        return {
            x: 20,  // 20px from left
            y: 20,  // 20px from top
        };
    },
});
```

## Accessing getBoundingClientRect

The `element` parameter is the focused `HTMLInputElement` or `HTMLTextAreaElement`. Use `getBoundingClientRect()` to get its position:

```typescript
type TextInputElement = HTMLInputElement | HTMLTextAreaElement;

function customPosition(element: TextInputElement): ToolbarPosition {
    const rect = element.getBoundingClientRect();
    
    // rect properties:
    // - rect.left: Distance from left edge of viewport
    // - rect.top: Distance from top edge of viewport
    // - rect.right: Distance from left edge to right side of element
    // - rect.bottom: Distance from top edge to bottom of element
    // - rect.width: Element width
    // - rect.height: Element height
    
    return {
        x: rect.left,
        y: rect.bottom + 5,
    };
}
```

## Styling the Toolbar Container

The `getToolbarProps()` function automatically applies positioning styles:

```typescript
const getToolbarProps = () => ({
    style: {
        left: toolbarState.position?.x ?? 0,
        position: 'fixed' as const,
        top: toolbarState.position?.y ?? 0,
        zIndex: 1000,
    },
    // ... event handlers
});
```

<Tip>
The toolbar uses `position: fixed` with `zIndex: 1000` to ensure it appears above other content. Adjust your CSS if you need different layering behavior.
</Tip>

## Responsive Positioning

Adjust positioning based on screen size:

```typescript
const { getInputProps } = useFormattingToolbar({
    getPosition: (element) => {
        const rect = element.getBoundingClientRect();
        const isMobile = window.innerWidth < 768;
        
        if (isMobile) {
            // On mobile, center the toolbar at the bottom of the screen
            return {
                x: (window.innerWidth - 300) / 2,
                y: window.innerHeight - 80,
            };
        }
        
        // On desktop, use default positioning
        return {
            x: rect.left,
            y: rect.bottom + 5,
        };
    },
});
```

## Update Position Dynamically

The position is calculated when the toolbar is shown. If you need to update it:

```typescript
import { useFormattingToolbar } from 'blumbaben';
import { useEffect } from 'react';

function DynamicPositioning() {
    const { getInputProps, showToolbar, toolbarState } = useFormattingToolbar({
        getPosition: (element) => {
            const rect = element.getBoundingClientRect();
            return { x: rect.left, y: rect.bottom + 5 };
        },
    });

    useEffect(() => {
        if (!toolbarState.isVisible || !toolbarState.activeElement) return;

        // Recalculate position on scroll or resize
        const handleUpdate = () => {
            showToolbar(toolbarState.activeElement!);
        };

        window.addEventListener('scroll', handleUpdate);
        window.addEventListener('resize', handleUpdate);

        return () => {
            window.removeEventListener('scroll', handleUpdate);
            window.removeEventListener('resize', handleUpdate);
        };
    }, [toolbarState.isVisible, toolbarState.activeElement, showToolbar]);

    return <textarea {...getInputProps()} />;
}
```

## Debugging Positioning

Add temporary visual indicators to debug positioning:

```typescript
const { getInputProps, getToolbarProps, isVisible } = useFormattingToolbar({
    getPosition: (element) => {
        const rect = element.getBoundingClientRect();
        const position = { x: rect.left, y: rect.bottom + 5 };
        
        // Debug: Log position
        console.log('Toolbar position:', position);
        console.log('Element rect:', rect);
        
        return position;
    },
});

return (
    <>
        <textarea {...getInputProps()} />
        {isVisible && (
            <div
                {...getToolbarProps()}
                style={{
                    ...getToolbarProps().style,
                    // Add visual debugging
                    border: '2px solid red',
                    backgroundColor: 'yellow',
                }}
            >
                Toolbar
            </div>
        )}
    </>
);
```
