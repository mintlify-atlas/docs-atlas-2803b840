# Markdown Formatting Toolbar

This example demonstrates a complete toolbar with common markdown formatting buttons. Users can select text and apply markdown formatting like **bold**, *italic*, `code`, links, and more.

## Complete Working Example

<CodeGroup>

```tsx Component
import React, { useState } from 'react';
import { useFormattingToolbar, FormattingToolbar } from 'blumbaben';

function MarkdownEditor() {
  const [content, setContent] = useState('');
  const { getInputProps } = useFormattingToolbar();

  return (
    <div className="editor-container">
      <textarea
        {...getInputProps()}
        value={content}
        onChange={(e) => setContent(e.target.value)}
        placeholder="Start typing markdown content..."
        className="markdown-textarea"
        rows={12}
      />

      <FormattingToolbar
        className="markdown-toolbar"
        style={{
          backgroundColor: 'white',
          border: '1px solid #e2e8f0',
          borderRadius: '8px',
          boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
          padding: '8px',
        }}
      >
        {(applyFormat) => (
          <div className="toolbar-buttons">
            {/* Text formatting */}
            <button
              onClick={() => applyFormat((text) => `**${text}**`)}
              className="format-btn"
              title="Bold"
            >
              <strong>B</strong>
            </button>

            <button
              onClick={() => applyFormat((text) => `*${text}*`)}
              className="format-btn"
              title="Italic"
            >
              <em>I</em>
            </button>

            <button
              onClick={() => applyFormat((text) => `~~${text}~~`)}
              className="format-btn"
              title="Strikethrough"
            >
              <s>S</s>
            </button>

            <div className="divider" />

            {/* Code formatting */}
            <button
              onClick={() => applyFormat((text) => `\`${text}\``)}
              className="format-btn"
              title="Inline Code"
            >
              {'<>'}
            </button>

            <button
              onClick={() => applyFormat((text) => `\`\`\`\n${text}\n\`\`\``)}
              className="format-btn"
              title="Code Block"
            >
              {'{ }'}
            </button>

            <div className="divider" />

            {/* Links and quotes */}
            <button
              onClick={() => applyFormat((text) => `[${text}](url)`)}
              className="format-btn"
              title="Link"
            >
              🔗
            </button>

            <button
              onClick={() => applyFormat((text) => `> ${text}`)}
              className="format-btn"
              title="Quote"
            >
              "
            </button>

            <div className="divider" />

            {/* List formatting */}
            <button
              onClick={() => applyFormat((text) => `- ${text}`)}
              className="format-btn"
              title="Bullet List"
            >
              •
            </button>

            <button
              onClick={() => applyFormat((text) => `1. ${text}`)}
              className="format-btn"
              title="Numbered List"
            >
              1.
            </button>

            <button
              onClick={() => applyFormat((text) => `- [ ] ${text}`)}
              className="format-btn"
              title="Task List"
            >
              ☑
            </button>
          </div>
        )}
      </FormattingToolbar>
    </div>
  );
}

export default MarkdownEditor;
```

## How It Works

1. **Textarea with Toolbar Integration**: The `getInputProps()` spread automatically adds focus/blur handlers to show/hide the toolbar
2. **Markdown Formatters**: Each button applies a specific markdown syntax wrapper around the selected text
3. **Formatter Functions**: Simple functions that wrap text with markdown syntax:
   - `**text**` for bold
   - `*text*` for italic
   - `` `text` `` for inline code
   - And more!

## Formatter Function Examples

Here are the individual formatter functions you can extract and reuse:

```tsx Formatter Functions
// Text styling
const makeBold = (text: string) => `**${text}**`;
const makeItalic = (text: string) => `*${text}*`;
const makeStrikethrough = (text: string) => `~~${text}~~`;

// Code formatting
const makeInlineCode = (text: string) => `\`${text}\``;
const makeCodeBlock = (text: string) => `\`\`\`\n${text}\n\`\`\``;

// Links and quotes
const makeLink = (text: string) => `[${text}](url)`;
const makeQuote = (text: string) => `> ${text}`;

// Lists
const makeBulletList = (text: string) => `- ${text}`;
const makeNumberedList = (text: string) => `1. ${text}`;
const makeTaskList = (text: string) => `- [ ] ${text}`;

// Headings
const makeH1 = (text: string) => `# ${text}`;
const makeH2 = (text: string) => `## ${text}`;
const makeH3 = (text: string) => `### ${text}`;
```

<Tip>
**Pro Tip**: If no text is selected, the formatter applies to the entire content of the textarea. This is perfect for quick transformations like converting everything to a code block!
</Tip>

## Advanced: Multi-line List Formatting

You can create more advanced formatters that work on multiple lines:

```tsx Multi-line Formatters
function MultiLineMarkdownEditor() {
  const [content, setContent] = useState('');
  const { getInputProps } = useFormattingToolbar();

  // Convert each line to a bullet point
  const makeMultiLineBulletList = (text: string) => {
    return text
      .split('\n')
      .map(line => line.trim() ? `- ${line}` : '')
      .join('\n');
  };

  // Convert each line to a numbered list
  const makeMultiLineNumberedList = (text: string) => {
    return text
      .split('\n')
      .map((line, index) => line.trim() ? `${index + 1}. ${line}` : '')
      .join('\n');
  };

  // Indent selected lines
  const indentLines = (text: string) => {
    return text
      .split('\n')
      .map(line => `  ${line}`)
      .join('\n');
  };

  return (
    <div>
      <textarea
        {...getInputProps()}
        value={content}
        onChange={(e) => setContent(e.target.value)}
        placeholder="Type multiple lines and format them together..."
        rows={10}
      />

      <FormattingToolbar>
        {(applyFormat) => (
          <div style={{ display: 'flex', gap: '8px', padding: '8px' }}>
            <button onClick={() => applyFormat(makeMultiLineBulletList)}>
              Multi-line Bullets
            </button>
            <button onClick={() => applyFormat(makeMultiLineNumberedList)}>
              Multi-line Numbers
            </button>
            <button onClick={() => applyFormat(indentLines)}>
              Indent Lines
            </button>
          </div>
        )}
      </FormattingToolbar>
    </div>
  );
}
```
