# Text Transformation Toolbar

This example demonstrates a toolbar with various text transformation utilities. Perfect for cleaning up user input, changing case, or reformatting text quickly.

## Complete Working Example

<CodeGroup>

```tsx Component
import React, { useState } from 'react';
import { useFormattingToolbar, FormattingToolbar } from 'blumbaben';

function TextTransformEditor() {
  const [text, setText] = useState('');
  const { getInputProps } = useFormattingToolbar();

  return (
    <div className="editor-wrapper">
      <div className="editor-header">
        <h2>Text Transformer</h2>
        <p>Select text and apply transformations, or transform the entire content</p>
      </div>

      <textarea
        {...getInputProps()}
        value={text}
        onChange={(e) => setText(e.target.value)}
        placeholder="Type or paste text here to transform it..."
        className="transform-textarea"
        rows={10}
      />

      <FormattingToolbar
        className="transform-toolbar"
        style={{
          backgroundColor: '#1e293b',
          border: '1px solid #334155',
          borderRadius: '12px',
          boxShadow: '0 10px 40px rgba(0, 0, 0, 0.3)',
          padding: '12px',
        }}
      >
        {(applyFormat) => (
          <div className="toolbar-content">
            <div className="toolbar-section">
              <span className="section-label">Case</span>
              <div className="button-group">
                <button
                  onClick={() => applyFormat((text) => text.toUpperCase())}
                  className="transform-btn"
                  title="Convert to UPPERCASE"
                >
                  ABC
                </button>

                <button
                  onClick={() => applyFormat((text) => text.toLowerCase())}
                  className="transform-btn"
                  title="Convert to lowercase"
                >
                  abc
                </button>

                <button
                  onClick={() => applyFormat((text) =>
                    text.replace(/\w\S*/g, (word) =>
                      word.charAt(0).toUpperCase() + word.slice(1).toLowerCase()
                    )
                  )}
                  className="transform-btn"
                  title="Convert to Title Case"
                >
                  Abc
                </button>

                <button
                  onClick={() => applyFormat((text) => {
                    // Sentence case: capitalize first letter of each sentence
                    return text.replace(/(^|[.!?]\s+)([a-z])/g, (match, separator, letter) =>
                      separator + letter.toUpperCase()
                    );
                  })}
                  className="transform-btn"
                  title="Convert to Sentence case"
                >
                  Aa
                </button>
              </div>
            </div>

            <div className="toolbar-divider" />

            <div className="toolbar-section">
              <span className="section-label">Clean</span>
              <div className="button-group">
                <button
                  onClick={() => applyFormat((text) => text.trim())}
                  className="transform-btn"
                  title="Remove leading/trailing whitespace"
                >
                  Trim
                </button>

                <button
                  onClick={() => applyFormat((text) => text.replace(/\s+/g, ' '))}
                  className="transform-btn"
                  title="Remove extra spaces"
                >
                  Spaces
                </button>

                <button
                  onClick={() => applyFormat((text) => text.replace(/\n/g, ' '))}
                  className="transform-btn"
                  title="Remove line breaks"
                >
                  Lines
                </button>

                <button
                  onClick={() => applyFormat((text) =>
                    text.replace(/[^a-zA-Z0-9\s]/g, '')
                  )}
                  className="transform-btn"
                  title="Remove special characters"
                >
                  A-Z
                </button>
              </div>
            </div>

            <div className="toolbar-divider" />

            <div className="toolbar-section">
              <span className="section-label">Format</span>
              <div className="button-group">
                <button
                  onClick={() => applyFormat((text) =>
                    text
                      .toLowerCase()
                      .replace(/[^a-z0-9]+/g, '-')
                      .replace(/(^-|-$)+/g, '')
                  )}
                  className="transform-btn"
                  title="Convert to URL slug"
                >
                  Slug
                </button>

                <button
                  onClick={() => applyFormat((text) =>
                    text
                      .toLowerCase()
                      .replace(/[^a-z0-9]+/g, '_')
                      .replace(/(^_|_$)+/g, '')
                  )}
                  className="transform-btn"
                  title="Convert to snake_case"
                >
                  snake_case
                </button>

                <button
                  onClick={() => applyFormat((text) =>
                    text
                      .replace(/[^a-zA-Z0-9]+(.)/g, (_, char) => char.toUpperCase())
                      .replace(/^./, (char) => char.toLowerCase())
                  )}
                  className="transform-btn"
                  title="Convert to camelCase"
                >
                  camelCase
                </button>

                <button
                  onClick={() => applyFormat((text) => {
                    const trimmed = text.trim();
                    return trimmed.length > 0 ? `"${trimmed}"` : trimmed;
                  })}
                  className="transform-btn"
                  title="Wrap in quotes"
                >
                  "..."
                </button>
              </div>
            </div>
          </div>
        )}
      </FormattingToolbar>
    </div>
  );
}

export default TextTransformEditor;
```

## How It Works

1. **Case Transformations**: Convert text between uppercase, lowercase, title case, and sentence case
2. **Text Cleaning**: Remove unwanted whitespace, line breaks, and special characters
3. **Format Conversions**: Convert text to URL slugs, snake_case, camelCase, or wrap in quotes
4. **Selection Support**: Works on selected text or entire content when nothing is selected

## Transformer Function Library

Here's a reusable library of text transformation functions:

```tsx Transformer Functions
// Case transformations
const toUpperCase = (text: string) => text.toUpperCase();

const toLowerCase = (text: string) => text.toLowerCase();

const toTitleCase = (text: string) =>
  text.replace(/\w\S*/g, (word) =>
    word.charAt(0).toUpperCase() + word.slice(1).toLowerCase()
  );

const toSentenceCase = (text: string) =>
  text.replace(/(^|[.!?]\s+)([a-z])/g, (match, separator, letter) =>
    separator + letter.toUpperCase()
  );

// Text cleaning
const trimWhitespace = (text: string) => text.trim();

const removeExtraSpaces = (text: string) => text.replace(/\s+/g, ' ');

const removeLineBreaks = (text: string) => text.replace(/\n/g, ' ');

const removeSpecialChars = (text: string) => text.replace(/[^a-zA-Z0-9\s]/g, '');

const removeNumbers = (text: string) => text.replace(/[0-9]/g, '');

// Format conversions
const toSlug = (text: string) =>
  text
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/(^-|-$)+/g, '');

const toSnakeCase = (text: string) =>
  text
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '_')
    .replace(/(^_|_$)+/g, '');

const toCamelCase = (text: string) =>
  text
    .replace(/[^a-zA-Z0-9]+(.)/g, (_, char) => char.toUpperCase())
    .replace(/^./, (char) => char.toLowerCase());

const toPascalCase = (text: string) =>
  text
    .replace(/[^a-zA-Z0-9]+(.)/g, (_, char) => char.toUpperCase())
    .replace(/^./, (char) => char.toUpperCase());

const toKebabCase = (text: string) =>
  text
    .replace(/([a-z])([A-Z])/g, '$1-$2')
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/(^-|-$)+/g, '');

// Text wrapping
const wrapInQuotes = (text: string) => {
  const trimmed = text.trim();
  return trimmed.length > 0 ? `"${trimmed}"` : trimmed;
};

const wrapInParentheses = (text: string) => `(${text})`;

const wrapInBrackets = (text: string) => `[${text}]`;

const wrapInBraces = (text: string) => `{${text}}`;
```

<Tip>
**Advanced Tip**: You can chain transformations by creating composite formatters:

```tsx
const cleanAndSlug = (text: string) => toSlug(removeExtraSpaces(trimWhitespace(text)));

applyFormat(cleanAndSlug);
```
</Tip>

## Advanced: Custom Transformations

Create domain-specific transformations for your use case:

```tsx Custom Transformations
// Email address formatter
const formatEmail = (text: string) =>
  text.toLowerCase().trim().replace(/\s+/g, '');

// Phone number formatter (US)
const formatPhoneUS = (text: string) => {
  const digits = text.replace(/\D/g, '');
  if (digits.length === 10) {
    return `(${digits.slice(0, 3)}) ${digits.slice(3, 6)}-${digits.slice(6)}`;
  }
  return text;
};

// Credit card formatter
const formatCreditCard = (text: string) => {
  const digits = text.replace(/\D/g, '');
  return digits.match(/.{1,4}/g)?.join(' ') || text;
};

// Currency formatter
const formatCurrency = (text: string) => {
  const number = parseFloat(text.replace(/[^0-9.]/g, ''));
  return isNaN(number) ? text : `$${number.toFixed(2)}`;
};

// Date formatter (MM/DD/YYYY to YYYY-MM-DD)
const formatDateISO = (text: string) => {
  const match = text.match(/(\d{1,2})\/(\d{1,2})\/(\d{4})/);
  if (match) {
    const [, month, day, year] = match;
    return `${year}-${month.padStart(2, '0')}-${day.padStart(2, '0')}`;
  }
  return text;
};

// Usage in toolbar
<FormattingToolbar>
  {(applyFormat) => (
    <div>
      <button onClick={() => applyFormat(formatEmail)}>Format Email</button>
      <button onClick={() => applyFormat(formatPhoneUS)}>Format Phone</button>
      <button onClick={() => applyFormat(formatCreditCard)}>Format Card</button>
      <button onClick={() => applyFormat(formatCurrency)}>Format $</button>
    </div>
  )}
</FormattingToolbar>
```
